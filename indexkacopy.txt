            function makeEditable(elementId, editIconId) {
                const element = document.getElementById(elementId);
                const editIcon = document.getElementById(editIconId);
                
                if (!element || !editIcon) return;

                // Set default text if empty
                if (!element.textContent.trim()) {
                    element.textContent = `Telecaller ${elementId.replace(/[^0-9]/g, '')}`;
                }

                editIcon.addEventListener('click', () => {
                    const currentText = element.textContent;
                    const input = document.createElement('input');
                    input.type = 'text';
                    input.value = currentText;
                    input.className = 'name-edit-input';
                    
                    element.textContent = '';
                    element.appendChild(input);
                    input.focus();

                    input.addEventListener('blur', () => {
                        element.textContent = input.value.trim() || currentText;
                    });

                    input.addEventListener('keypress', (e) => {
                        if (e.key === 'Enter') {
                            element.textContent = input.value.trim() || currentText;
                        }
                    });
                });
            }

        // Function to update activity counts
        async function updateActivityCounts() {
            try {
                const telecallerEmail = localStorage.getItem('telecallerEmail');
                const telecallerName = telecallerEmail.split('@')[0];
                const db = firebase.database();
                
                const snapshot = await db.ref(telecallerName).once('value');
                const calls = snapshot.val() || {};

                let counts = {
                    unanswered: 0,
                    interested: 0,
                    callback: 0
                };

                Object.values(calls).forEach(call => {
                    if (call.status === 'Unanswered') counts.unanswered++;
                    if (call.status === 'Interested') counts.interested++;
                    if (call.status === 'Callback') counts.callback++;
                });

                // Update count displays
                optionsModal.querySelector('#unansweredCount').textContent = `(${counts.unanswered})`;
                optionsModal.querySelector('#interestedCount').textContent = `(${counts.interested})`;
                optionsModal.querySelector('#callbackCount').textContent = `(${counts.callback})`;

            } catch (error) {
                console.error('Error updating counts:', error);
            }
        }
